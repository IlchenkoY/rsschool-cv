## Code Examples

function createChunks(arr, chunkSize) {
  function chunkify(inputArray, size) {
    return inputArray.length === 0
      ? []
      : [inputArray.slice(0, size)].concat(
          chunkify(inputArray.slice(size), size)
        );
  }

  return chunkify(arr, chunkSize);
}
